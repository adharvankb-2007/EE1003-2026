% Question
\begin{frame}{Question}
For real $x$, the maximum value of
\[
\frac{e^{\sin x}}{e^{\cos x}}
\]
is:

\begin{enumerate}[label=\alph*)]
    \item $1$
    \item $e$
    \item $e^{\sqrt{2}}$
    \item $\infty$
\end{enumerate}
\end{frame}

% Step 1
\begin{frame}{Simplification}
\[
\frac{e^{\sin x}}{e^{\cos x}} = e^{\sin x - \cos x}
\]

Thus, we need to maximize:
\[
\sin x - \cos x
\]
\end{frame}

% Step 2
\begin{frame}{Trigonometric Transformation}
Using identity:
\[
a\sin x + b\cos x = \sqrt{a^2 + b^2}\,\sin(x + \phi)
\]

\[
\sin x - \cos x = \sqrt{2}\,\sin\left(x - \frac{\pi}{4}\right)
\]
\end{frame}

% Step 3
\begin{frame}{Maximum Value}
Since:
\[
-1 \leq \sin(\theta) \leq 1
\]

\[
\max(\sin x - \cos x) = \sqrt{2}
\]

Therefore:
\[
\max e^{\sin x - \cos x} = e^{\sqrt{2}}
\]
\end{frame}

% Final Answer
\begin{frame}{Final Answer}
\[
\boxed{e^{\sqrt{2}}}
\]

\begin{itemize}
    \item Correct option: \textbf{(c)}
\end{itemize}
\end{frame}
